2.2.1 什么是 Linux Cgroups

kernel 接口

＃ 挂载一个 hierarchy
sudo mount -t cgroup - o none ,name=cgroup-test cgroup-test ./cgroup-test

Linux 挂载命令通用格式：mount -t 文件系统类型 -o 挂载选项 挂载源 目标挂载点

mount：挂载命令，将「文件系统 / 虚拟设备」关联到系统的一个目录下

-t cgroup：-t = type，指定要挂载的文件系统类型为 cgroup。Cgroup 是 Linux 内核提供的虚拟文件系统（不占用真实磁盘，只存在于内存中，用于内核和用户态交互）

-o none,name=cgroup-test：-o = options，挂载参数（多个参数用逗号分隔）：

  • none 表示这个 Cgroup 层级不绑定任何具体资源子系统（子系统 = CPU / 内存 / 磁盘 IO 等控制器，先创建一个空层级，不限制任何资源）
  • name=cgroup-test 给这个 Cgroup 层级起一个唯一名字 cgroup-test，方便内核识别管理
  • cgroup-test 挂载源。对 Cgroup 虚拟文件系统来说，源就是层级的名字（和上面 name= 对应），不是真实磁盘设备

./cgroup-test：目标挂载点

mount | grep memory
# cgroup on /sys/fs/cgroup/memory type cgroup (rw , nosuid,nodev , noexec ,relatime ,memory , nsroot=/)

mount 列出系统中所有已挂载的文件系统（包括真实磁盘、虚拟文件系统，比如 Cgroup）

输出片段 含义
cgroup 挂载的文件系统类型：Cgroup 虚拟文件系统
on /sys/fs/cgroup/memory 挂载点：系统默认的 Cgroup 内存控制器路径（Linux 自动创建，不是手动建的）
type cgroup 再次确认：这是 Cgroup 虚拟文件系统
(rw,...) 挂载参数

挂载参数详解

rw：读写权限（必须可读写，才能配置内存限制）

nosuid、nodev、noexec：安全参数：禁止提权、禁止设备文件、禁止执行程序（Cgroup 目录只用于配置，不需要这些功能）
relatime：性能优化：文件时间戳更新机制
memory：这个 Cgroup 层级专门绑定了内存子系统，专门管理进程内存（限制内存用量、统计内存使用）
nsroot=/：内核命名空间参数，普通使用无需关注

cgroup V2 与 V1 的关键区别

cgroup V1 的独立子系统目录如 /sys/fs/cgroup/memory，但在 cgroup V2 中：

  • 所有控制器统一挂载在/sys/fs/cgroup，无独立子系统目录
  • 内存限制通过 memory.max、memory.high等文件设置，而非 V1 的 memory.limit_in_bytes
  • 启用控制器需通过cgroup.subtree_control配置，而非自动挂载子系统
基于 memory subsystem 的进程内存限制
# 首先，在不做限制的情况下，启动一个占用内存的 stress 进程
[memory] stress --vm-bytes 200m --vm-keep -m 1

# 创建一个 cgroup
[memory] sudo mkdir test-limit-memory && cd test-limit-memory

# 设置最大 cgroup 的最大内存占用为 lOOMB
[test-limit-memory] sudo sh -c " echo  100m > memory.limit_in_bytes"

# 将当前进程移动到这个 cgroup 中
[test-limit-memory] sudo sh -c "echo $$ > tasks"

# 再次运行占用内存 200MB 的 stress j进程
[test-limit-memory] stress --vm-bytes 200m --vm-keep -m 1

上述是基于 cgroup V1 的版本，等价改造为 cgroup V2 版本：

# 进入cgroup根目录
cd /sys/fs/cgroup

# 根目录仅启用 1 次 memory 控制器（永久生效）
echo "+memory" | sudo tee cgroup.subtree_control

# 创建测试目录
sudo mkdir test-limit-memory && cd test-limit-memory

# 设置内存上限 100MB
echo 100m | sudo tee memory.max

# 当前进程加入 test-limit-memory 层级。只有叶子 cgroup 才能添加进程
echo $$ | sudo tee cgroup.procs

# 测试：启动 200MB 内存进程（会被限制）
stress --vm-bytes 200m --vm-keep -m 1

# 测试结束：Ctrl + C 终止 stress 进程，当前进程移回 cgroup 根层级
echo $$ | sudo tee /sys/fs/cgroup/cgroup.procs

# 回到根目录
cd ..

# 删除测试目录
sudo rmdir test-limit-memory

2.2.2 Docker 是如何使用 Cgroups 的

docker run -m 设置内存限制：

# 启动内存限制 128M 的容器
sudo docker run -itd m 128m ubuntu
# <container-id>: 957459145e9092618837cf94alcb356e206f2f0da560b40cb31035e442d3dfll

# docker 会为每个容器在系统的 hierarchy 中创建 cgroup
cd /sys/fs/cgroup/memory/docker/<container-id>

# 查看内存限制
cat memory.limit_in_bytes
# 134217728

# 查看当前内存占用
cat memory.usage_in-bytes
# 430080

上述是基于 cgroup V1 的版本，等价改造为 cgroup V2 版本：

# 启动内存限制 128M 的容器
sudo docker run -itd --rm -m 128m ubuntu

# 获取容器 ID
docker ps --no-trunc
# <container-id>

# 进入 cgroup V2 对应目录
cd /sys/fs/cgroup/system.slice/docker-<container-id>.scope/

# 查看内存限制（验证 128M）
cat memory.max

# 查看当前内存
cat memory.current

2.2.3 用 Go 语言实现通过 cgroup 限制容器的资源

把 demo 从 cgroup V1 版改成 cgroup V2 版：

package main

import (
	"os/exec"
	"path"
	"os"
	"fmt"
	"io/ioutil"
	"syscall"
	"strconv"
)

const cgroupV2HirearchyMount = "/sys/fs/cgroup"
const myCgroupName = "test-memory-limit"

func main() {
	if os.Args[0] == "/proc/self/exe" {
		/* 子进程 */
		fmt.Printf("current pid %d\n", syscall.Getpid())

		cmd := exec.Command("stress", "--vm-bytes", "200m", "--vm-keep", "-m", "1")
		cmd.Stdin = os.Stdin
		cmd.Stdout = os.Stdout
		cmd.Stderr = os.Stderr

		if err := cmd.Run(); err != nil {
			fmt.Println("[sub-process ERROR]", err)
			os.Exit(1)
		}
	}

	/* 父进程启动子进程 */
	cmd := exec.Command("/proc/self/exe")
	cmd.SysProcAttr = &syscall.SysProcAttr {
		Cloneflags : syscall.CLONE_NEWUTS |
						syscall.CLONE_NEWPID |
						syscall.CLONE_NEWNS |
						syscall.CLONE_NEWUSER,
	}
	cmd.Stdin = os.Stdin
	cmd.Stdout = os.Stdout
	cmd.Stderr = os.Stderr

	/* 新建 cgroup 的路径 */
	testCgroupPath := path.Join(cgroupV2HirearchyMount, myCgroupName)

	if err := cmd.Start(); err != nil {
		fmt.Println("ERROR", err)
		os.Exit(1)
	} else {
		/* 得到子进程映射在外部命名空间的 PID */
		fmt.Printf("forked process's PID: %v\n", cmd.Process.Pid)

		/* 在 cgroup V2 的 hierarchy 中创建测试 cgroup: test-memory-limit */
		if err := os.Mkdir(testCgroupPath, 0755); err != nil {
			fmt.Printf("Create cgroups failed: %v\n", err)
			os.Exit(1)
		}

		/* 将容器进程加载到新建的 cgroup */
		if err := ioutil.WriteFile(path.Join(testCgroupPath, "cgroup.procs"),
									[]byte(strconv.Itoa(cmd.Process.Pid)),
									0644); err != nil {
			fmt.Printf("Move sub-process failed: %v\n", err)
			os.Exit(1)
		}
		
		/* 对新建的 cgroup 进行内存限制 */
		if err := ioutil.WriteFile(path.Join(testCgroupPath, "memory.max"),
									[]byte("100m"),
									0644); err != nil {
			fmt.Printf("Set memory limit failed: %v\n", err)
			os.Exit(1)
		}
	}

	/* 等待子进程退出 */
	cmd.Process.Wait()

	os.Remove(testCgroupPath)
}
